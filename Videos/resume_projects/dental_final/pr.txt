    <script>
        function toggleTextFields(checkbox) {
            const textFieldsContainer = checkbox.parentElement.nextElementSibling;
                        textFieldsContainer.style.display = checkbox.checked ? 'block' : 'none';
            
        }
        

        const selectedNumbers = [];
        document.querySelectorAll('.number').forEach(number => {
            number.addEventListener('click', () => {
                if (number.classList.contains('highlighted')) {
                    number.classList.remove('highlighted');
                    const index = selectedNumbers.indexOf(number.textContent);
                    if (index > -1) {
                        selectedNumbers.splice(index, 1);
                    }
                } else {
                    number.classList.add('highlighted');
                    selectedNumbers.push(number.textContent);
                }
                document.getElementById('selected-numbers').textContent = 'Selected Numbers: ' + selectedNumbers.join(', ');
            });
        });


    </script>
